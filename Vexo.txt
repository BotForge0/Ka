local function safeCall(fn, ...)
    local ok, err = pcall(fn, ...)
    if not ok then
        warn("[BRM5 Error]: " .. tostring(err))
    end
    return ok
end

local function addConnection(name, conn)
    if not Connections[name] then
        Connections[name] = {}
    end
    table.insert(Connections[name], conn)
end

local function clearConnections(name)
    if Connections[name] then
        for _, c in ipairs(Connections[name]) do
            pcall(function() c:Disconnect() end)
        end
        Connections[name] = {}
    end
end

local function isEnemy(obj)
    if not obj or not obj:IsA("Model") then return false end
    if obj == LocalPlayer.Character then return false end
    local hum = obj:FindFirstChildOfClass("Humanoid")
    if not hum then return false end
    if hum.Health <= 0 then return false end
    return true
end

local function getEnemies()
    local enemies = {}
    for _, obj in pairs(Workspace:GetDescendants()) do
        if isEnemy(obj) then
            table.insert(enemies, obj)
        end
    end
    return enemies
end

local function getClosestEnemyToMouse()
    local closest = nil
    local closestDist = Config.SilentAimRadius

    for _, enemy in pairs(getEnemies()) do
        local head = enemy:FindFirstChild("Head")
        if head then
            local screenPos, onScreen = Camera:WorldToScreenPoint(head.Position)
            if onScreen then
                local mousePos = Vector2.new(Mouse.X, Mouse.Y)
                local targetPos = Vector2.new(screenPos.X, screenPos.Y)
                local dist = (mousePos - targetPos).Magnitude
                if dist < closestDist then
                    closestDist = dist
                    closest = enemy
                end
            end
        end
    end
    return closest
end

local function applyFullbright()
    if applying then return end
    applying = true
    safeCall(function()
        Lighting.Brightness     = Config.Brightness
        Lighting.ClockTime      = Config.ClockTime
        Lighting.Ambient        = Config.Ambient
        Lighting.OutdoorAmbient = Config.OutdoorAmbient
        Lighting.GlobalShadows  = Config.GlobalShadows
        Lighting.FogStart       = 999999
        Lighting.FogEnd         = 999999

        for _, child in ipairs(Lighting:GetChildren()) do
            if child:IsA("Atmosphere") then
                child.Density = 0
            end
        end
    end)
    applying = false
end

local function enableFullbright()
    States.fullbright = true
    applyFullbright()

    local props = {"Brightness", "ClockTime", "Ambient", "OutdoorAmbient", "GlobalShadows", "FogStart", "FogEnd"}
    for _, prop in ipairs(props) do
        addConnection("fullbright", Lighting:GetPropertyChangedSignal(prop):Connect(function()
            if States.fullbright then applyFullbright() end
        end))
    end
end

local function disableFullbright()
    States.fullbright = false
    clearConnections("fullbright")
    safeCall(function()
        applying = true
        Lighting.Brightness     = Original.Brightness
        Lighting.ClockTime      = Original.ClockTime
        Lighting.Ambient        = Original.Ambient
        Lighting.OutdoorAmbient = Original.OutdoorAmbient
        Lighting.GlobalShadows  = Original.GlobalShadows
        Lighting.FogStart       = Original.FogStart
        Lighting.FogEnd         = Original.FogEnd
        applying = false
    end)
end

local function applyESP(obj)
    safeCall(function()
        if not isEnemy(obj) then return end
        if obj:FindFirstChild("BRM5_ESP") then return end

        local hl = Instance.new("Highlight")
        hl.Name                = "BRM5_ESP"
        hl.Adornee             = obj
        hl.FillColor           = Config.ESP_FillColor
        hl.FillTransparency    = Config.ESP_FillTransparency
        hl.OutlineColor        = Config.ESP_OutlineColor
        hl.OutlineTransparency = Config.ESP_OutlineTransparency
        hl.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
        hl.Parent              = obj
    end)
end

local function enableESP()
    States.esp = true

    for _, obj in pairs(Workspace:GetDescendants()) do
        applyESP(obj)
    end

    addConnection("esp", Workspace.DescendantAdded:Connect(function(obj)
        if not States.esp then return end
        task.defer(function()
            task.wait(0.3)
            if obj and obj.Parent then
                applyESP(obj)
                if obj.Parent and obj.Parent:IsA("Model") then
                    applyESP(obj.Parent)
                end
            end
        end)
    end))
end

local function disableESP()
    States.esp = false
    clearConnections("esp")
    for _, obj in pairs(Workspace:GetDescendants()) do
        if obj:IsA("Highlight") and obj.Name == "BRM5_ESP" then
            pcall(function() obj:Destroy() end)
        end
    end
end

--  HITBOX
-- ══════════════════════════════════════

local function applyHitbox(model)
    safeCall(function()
        if not isEnemy(model) then return end
        local head = model:FindFirstChild("Head")
        if not head then return end
        if head:FindFirstChild("BRM5_Hitbox") then return end

        local box = Instance.new("Part")
        box.Name = "BRM5_Hitbox"
        box.Size = Vector3.new(Config.HitboxSize, Config.HitboxSize, Config.HitboxSize)
        box.Anchored = false
        box.CanCollide = false
        box.Massless = true
        box.Transparency = Config.HitboxTransparency
        box.BrickColor = BrickColor.new("Really red")
        box.Material = Enum.Material.ForceField
        box.Parent = head

        local weld = Instance.new("Weld")
        weld.Part0 = head
        weld.Part1 = box
        weld.C0 = CFrame.new()
        weld.Parent = box
    end)
end

local function enableHitbox()
    States.hitbox = true

    for _, obj in pairs(Workspace:GetDescendants()) do
        if obj:IsA("Model") then
            applyHitbox(obj)
        end
    end

    addConnection("hitbox", Workspace.DescendantAdded:Connect(function(obj)
        if not States.hitbox then return end
        task.defer(function()
            task.wait(0.5)
            if obj and obj:IsA("Model") then
                applyHitbox(obj)
            elseif obj and obj.Parent and obj.Parent:IsA("Model") then
                applyHitbox(obj.Parent)
            end
        end)
    end))

    -- Periodisch neue Feinde scannen (langsam, nicht jeden Frame)
    addConnection("hitbox", RunService.Heartbeat:Connect(function()
        if not States.hitbox then return end
    end))

    -- Langsamer Scanner alle 2 Sekunden
    task.defer(function()
        while States.hitbox do
            for _, obj in pairs(Workspace:GetDescendants()) do
                if obj:IsA("Model") then
                    applyHitbox(obj)
                end
            end
            task.wait(2)
        end
    end)
end

local function disableHitbox()
    States.hitbox = false
    clearConnections("hitbox")
    for _, obj in pairs(Workspace:GetDescendants()) do
        if obj:IsA("Part") and obj.Name == "BRM5_Hitbox" then
            pcall(function() obj:Destroy() end)
        end
    end
end

--  SILENT AIM
-- ══════════════════════════════════════

local oldIndex = nil
local oldNamecall = nil

local function enableSilentAim()
    States.silentaim = true

    safeCall(function()
        if hookmetamethod then
            if not oldIndex then
                oldIndex = hookmetamethod(game, "__index", newcclosure(function(self, key)
                    if States.silentaim and self == Mouse then
                        local target = getClosestEnemyToMouse()
                        if target then
                            local head = target:FindFirstChild("Head")
                            if head then
                                if key == "Hit" then
                                    return head.CFrame
                                elseif key == "Target" then
                                    return head
                                end
                            end
                        end
                    end
                    return oldIndex(self, key)
                end))
            end

            if not oldNamecall then
                oldNamecall = hookmetamethod(game, "__namecall", newcclosure(function(self, ...)
                    local method = getnamecallmethod()
                    if States.silentaim and (method == "FindPartOnRayWithIgnoreList" or method == "Raycast") then
                        local target = getClosestEnemyToMouse()
                        if target then
                            local head = target:FindFirstChild("Head")
                            if head and method == "Raycast" then
                                local args = {...}
                                if typeof(args[1]) == "Vector3" then
                                    local dir = (head.Position - args[1]).Unit * 1000
                                    args[2] = dir
                                    return oldNamecall(self, unpack(args))
                                end
                            end
                        end
                    end
                    return oldNamecall(self, ...)
                end))
            end
        else
            warn("[Silent Aim] hookmetamethod nicht verfügbar - Executor unterstützt es nicht")
        end
    end)
end

local function disableSilentAim()
    States.silentaim = false
end

--  AUTOSHOOT
-- ══════════════════════════════════════

local function enableAutoShoot()
    States.autoshoot = true

    task.defer(function()
        while States.autoshoot do
            safeCall(function()
                local target = getClosestEnemyToMouse()
                if target then
                    local head = target:FindFirstChild("Head")
                    if head then
                        local _, onScreen = Camera:WorldToScreenPoint(head.Position)
                        if onScreen then
                            if mouse1press and mouse1release then
                                mouse1press()
                                task.wait(0.03)
                                mouse1release()
                            end
                        end
                    end
                end
            end)
            task.wait(Config.AutoShootDelay)
        end
    end)
end

local function disableAutoShoot()
    States.autoshoot = false
end

--  FLY
-- ══════════════════════════════════════

local function enableFly()
    local char = LocalPlayer.Character
    if not char then return end
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then return end

    States.fly = true

    safeCall(function()
        flyBody = Instance.new("BodyVelocity")
        flyBody.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
        flyBody.Velocity = Vector3.zero
        flyBody.Parent = root

        flyGyro = Instance.new("BodyGyro")
        flyGyro.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
        flyGyro.D = 200
        flyGyro.P = 40000
        flyGyro.Parent = root
    end)

    addConnection("fly", RunService.Heartbeat:Connect(function()
        if not States.fly then return end
        if not root or not root.Parent then return end

        local dir = Vector3.zero
        local camCF = Camera.CFrame

        if UserInputService:IsKeyDown(Enum.KeyCode.W) then
            dir = dir + camCF.LookVector
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then
            dir = dir - camCF.LookVector
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then
            dir = dir - camCF.RightVector
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then
            dir = dir + camCF.RightVector
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
            dir = dir + Vector3.new(0, 1, 0)
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
            dir = dir - Vector3.new(0, 1, 0)
        end

        if dir.Magnitude > 0 then
            dir = dir.Unit
        end

        if flyBody then flyBody.Velocity = dir * Config.FlySpeed end
        if flyGyro then flyGyro.CFrame = camCF end
    end))
end

local function disableFly()
    States.fly = false
    clearConnections("fly")
    if flyBody then pcall(function() flyBody:Destroy() end) flyBody = nil end
    if flyGyro then pcall(function() flyGyro:Destroy() end) flyGyro = nil end
end

local function enablePotato()
    States.potato = true

    safeCall(function()
        settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
    end)

    local count = 0
    for _, obj in pairs(Workspace:GetDescendants()) do
        if count > 5000 then task.wait() count = 0 end -- Absturz vermeiden
        count = count + 1

        safeCall(function()
            if obj:IsA("Decal") or obj:IsA("Texture") then
                potatoOriginals[obj] = {Transparency = obj.Transparency}
                obj.Transparency = 1
            elseif obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") then
                potatoOriginals[obj] = {Enabled = obj.Enabled}
                obj.Enabled = false
            elseif obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") then
                potatoOriginals[obj] = {Enabled = obj.Enabled}
                obj.Enabled = false
            elseif obj:IsA("MeshPart") then
                potatoOriginals[obj] = {Material = obj.Material, TextureID = obj.TextureID}
                obj.Material = Enum.Material.SmoothPlastic
                obj.TextureID = ""
            elseif obj:IsA("BasePart") and not obj:IsA("MeshPart") then
                potatoOriginals[obj] = {Material = obj.Material}
                obj.Material = Enum.Material.SmoothPlastic
            end
        end)
    end

    for _, effect in pairs(Lighting:GetDescendants()) do
        safeCall(function()
            if effect:IsA("PostEffect") then
                potatoOriginals[effect] = {Enabled = effect.Enabled}
                effect.Enabled = false
            end
        end)
    end

    safeCall(function()
        local terrain = Workspace:FindFirstChildOfClass("Terrain")
        if terrain then
            terrain.WaterWaveSize = 0
            terrain.WaterWaveSpeed = 0
            pcall(function() sethiddenproperty(terrain, "Decoration", false) end)
        end
    end)
end

local function disablePotato()
    States.potato = false

    for obj, data in pairs(potatoOriginals) do
        safeCall(function()
            if obj and obj.Parent then
                for prop, val in pairs(data) do
                    obj[prop] = val
                end
            end
        end)
    end
    potatoOriginals = {}

    safeCall(function()
        settings().Rendering.QualityLevel = Enum.QualityLevel.Automatic
    end)
end

        knob.BorderSizePixel = 0
        knob.Parent = toggleFrame
        Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

        local clickBtn = Instance.new("TextButton")
        clickBtn.Size = UDim2.new(1, 0, 1, 0)
        clickBtn.BackgroundTransparency = 1
        clickBtn.Text = ""
        clickBtn.Parent = card

        clickBtn.MouseButton1Click:Connect(function()
            active = not active

            local goalPos = active and UDim2.new(1, -20, 0.5, -9) or UDim2.new(0, 2, 0.5, -9)
            local goalColor = active and Colors.toggleOn or Colors.toggleOff

            TweenService:Create(knob, TweenInfo.new(0.2), {Position = goalPos}):Play()
            TweenService:Create(toggleFrame, TweenInfo.new(0.2), {BackgroundColor3 = goalColor}):Play()

            if callback then
                safeCall(callback, active)
            end
        end)

        return function()
            active = not active
            local goalPos = active and UDim2.new(1, -20, 0.5, -9) or UDim2.new(0, 2, 0.5, -9)
            local goalColor = active and Colors.toggleOn or Colors.toggleOff
            TweenService:Create(knob, TweenInfo.new(0.2), {Position = goalPos}):Play()
            TweenService:Create(toggleFrame, TweenInfo.new(0.2), {BackgroundColor3 = goalColor}):Play()
            if callback then safeCall(callback, active) end
        end
    end

    local function createSlider(parent, text, order, min, max, default, callback)
        local card = Instance.new("Frame")
        card.Size = UDim2.new(1, 0, 0, 55)
        card.BackgroundColor3 = Colors.card
        card.BorderSizePixel = 0
        card.LayoutOrder = order
        card.Parent = parent
        Instance.new("UICorner", card).CornerRadius = UDim.new(0, 8)

        local label = Instance.new("TextLabel")
        label.Size = UDim2.new(0.7, 0, 0, 22)
        label.Position = UDim2.new(0, 12, 0, 4)
        label.BackgroundTransparency = 1
        label.Text = text
        label.TextColor3 = Colors.text
        label.TextSize = 12
        label.Font = Enum.Font.Gotham
        label.TextXAlignment = Enum.TextXAlignment.Left
        label.Parent = card

        local valueLabel = Instance.new("TextLabel")
        valueLabel.Size = UDim2.new(0.3, -12, 0, 22)
        valueLabel.Position = UDim2.new(0.7, 0, 0, 4)
        valueLabel.BackgroundTransparency = 1
        valueLabel.Text = tostring(default)
        valueLabel.TextColor3 = Colors.accent
        valueLabel.TextSize = 13
        valueLabel.Font = Enum.Font.GothamBold
        valueLabel.TextXAlignment = Enum.TextXAlignment.Right
        valueLabel.Parent = card

        -- Slider Track
        local track = Instance.new("Frame")
        track.Size = UDim2.new(1, -24, 0, 8)
        track.Position = UDim2.new(0, 12, 0, 35)
        track.BackgroundColor3 = Colors.sliderBg
        track.BorderSizePixel = 0
        track.Parent = card
        Instance.new("UICorner", track).CornerRadius = UDim.new(1, 0)

        local fill = Instance.new("Frame")
        local pct = (default - min) / (max - min)
        fill.Size = UDim2.new(pct, 0, 1, 0)
        fill.BackgroundColor3 = Colors.sliderFill
        fill.BorderSizePixel = 0
        fill.Parent = track
        Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)

        local knob = Instance.new("Frame")
        knob.Size = UDim2.new(0, 14, 0, 14)
        knob.Position = UDim2.new(pct, -7, 0.5, -7)
        knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        knob.BorderSizePixel = 0
        knob.ZIndex = 3
        knob.Parent = track
        Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

        -- Slider Input
        local dragging = false
        local clickArea = Instance.new("TextButton")
        clickArea.Size = UDim2.new(1, 0, 0, 20)
        clickArea.Position = UDim2.new(0, 0, 0, -6)
        clickArea.BackgroundTransparency = 1
        clickArea.Text = ""
        clickArea.ZIndex = 5
        clickArea.Parent = track

        local function updateSlider(inputX)
            local trackPos = track.AbsolutePosition.X
            local trackSize = track.AbsoluteSize.X
            local relX = math.clamp((inputX - trackPos) / trackSize, 0, 1)
            local value = math.floor(min + (max - min) * relX)

            fill.Size = UDim2.new(relX, 0, 1, 0)
            knob.Position = UDim2.new(relX, -7, 0.5, -7)
            valueLabel.Text = tostring(value)

            if callback then
                safeCall(callback, value)
            end
        end

        clickArea.MouseButton1Down:Connect(function()
            dragging = true
            updateSlider(Mouse.X)
        end)

        UserInputService.InputChanged:Connect(function(input)
            if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                updateSlider(input.Position.X)
            end
        end)

        UserInputService.InputEnded:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                dragging = false
            end
        end)
    end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if UserInputService:GetFocusedTextBox() then return end

    if input.KeyCode == Enum.KeyCode.I then
        UI.toggles.fullbright()
    elseif input.KeyCode == Enum.KeyCode.U then
        UI.toggles.esp()
    elseif input.KeyCode == Enum.KeyCode.H then
        UI.toggles.hitbox()
    elseif input.KeyCode == Enum.KeyCode.J then
        UI.toggles.silentaim()
    elseif input.KeyCode == Enum.KeyCode.K then
        UI.toggles.autoshoot()
    elseif input.KeyCode == Enum.KeyCode.F then
        UI.toggles.fly()
    elseif input.KeyCode == Enum.KeyCode.P then
        UI.toggles.potato()
    elseif input.KeyCode == Enum.KeyCode.T then
        UI.main.Visible = not UI.main.Visible
    end
end)

LocalPlayer.CharacterAdded:Connect(function()
    task.wait(1.5)

    if States.fly then
        disableFly()
        task.wait(0.3)
        enableFly()
    end

    if States.esp then
        task.defer(function()
            task.wait(0.5)
            for _, obj in pairs(Workspace:GetDescendants()) do
                if obj:IsA("Model") then applyESP(obj) end
            end
        end)
    end

    if States.hitbox then
        task.defer(function()
            task.wait(0.5)
            for _, obj in pairs(Workspace:GetDescendants()) do
                if obj:IsA("Model") then applyHitbox(obj) end
            end
        end)
    end
end)

LocalPlayer.CharacterRemoving:Connect(function()
    if flyBody then pcall(function() flyBody:Destroy() end) flyBody = nil end
    if flyGyro then pcall(function() flyGyro:Destroy() end) flyGyro = nil end
end)
