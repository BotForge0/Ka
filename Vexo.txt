-- MOUSE CLICK COMPAT
local function clickDown()
    if mouse1press then
        mouse1press()
    elseif EXE.syn then
        pcall(function() syn.input.mouse1down() end)
    else
        -- Fallback: VirtualInputManager
        pcall(function()
            game:GetService("VirtualInputManager"):SendMouseButtonEvent(
                Mouse.X, Mouse.Y, 0, true, game, 0)
        end)
    end
end

        mouse1release()
    elseif EXE.syn then
        pcall(function() syn.input.mouse1up() end)
    else
        pcall(function()
            game:GetService("VirtualInputManager"):SendMouseButtonEvent(

local function getChar(obj)
    -- Suche Character von Players
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LP and p.Character == obj then
            return p.Character
        end
    end
    return nil
end

local function isEnemy(model)
    if not model then return false end
    if not model:IsA("Model") then return false end
    if model == LP.Character then return false end
    
    local hum = model:FindFirstChildOfClass("Humanoid")
    if not hum then return false end
    if hum.Health <= 0 then return false end
    
    -- Muss Head oder HumanoidRootPart haben
    if not (model:FindFirstChild("Head") or model:FindFirstChild("HumanoidRootPart")) then
        return false
    end
    
    return true
end

local function getEnemies()
    local list = {}
    
    -- Zuerst Player Characters (beste Methode)
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LP and p.Character and isEnemy(p.Character) then
            table.insert(list, p.Character)
        end
    end
    
    -- Dann NPCs im Workspace
    for _, obj in ipairs(WS:GetChildren()) do
        if obj:IsA("Model") and isEnemy(obj) then
            -- Prüfe ob nicht schon als Player drin
            local already = false
            for _, p in ipairs(Players:GetPlayers()) do
                if p.Character == obj then already = true; break end
            end
            if not already then
                table.insert(list, obj)
            end
        end
    end
    
    return list
end

local function getBest()
    local best, bestD = nil, CFG.FOV
    local cx = Camera.ViewportSize.X / 2
    local cy = Camera.ViewportSize.Y / 2
    
    for _, e in ipairs(getEnemies()) do
        local part = e:FindFirstChild(CFG.AimPart)
                  or e:FindFirstChild("Head")
                  or e:FindFirstChild("HumanoidRootPart")
        if part then
            local tp = part.Position
            -- Prediction
            pcall(function()
                local vel = part.AssemblyLinearVelocity
                tp = tp + vel * CFG.Prediction
            end)
            
            local sp, vis = Camera:WorldToScreenPoint(tp)
            if vis and sp.Z > 0 then
                local dist = math.sqrt((sp.X-cx)^2 + (sp.Y-cy)^2)
                if dist < bestD then
                    bestD = dist
                    best  = {
                        model  = e,
                        part   = part,
                        pos    = tp,
                        screen = Vector2.new(sp.X, sp.Y),
                        dist   = bestD,
                    }
                end
            end
        end
    end
    return best
end

-- ══════════════════════

local fpsBackup  = {}
local fpsRunning = false

local function enableFPS()
    ST.fpsboost  = true
    fpsRunning   = true
    fpsBackup    = {}
    
    -- Level 1: Basis
    pcall(function()
        settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
    end)
    
    pcall(function()
        Lighting.GlobalShadows = false
        Lighting.FogEnd   = 9e9
        Lighting.FogStart = 9e9
        local t = WS:FindFirstChildOfClass("Terrain")
        if t then
            t.WaterWaveSize    = 0
            t.WaterWaveSpeed   = 0
            t.WaterReflectance = 0
        end
    end)
    
    -- Level 2: Post Processing
    if CFG.FPSLevel >= 2 then
        for _, e in ipairs(Lighting:GetDescendants()) do
            pcall(function()
                if e:IsA("PostEffect") then
                    fpsBackup[e] = {Enabled = e.Enabled}
                    e.Enabled = false
                elseif e:IsA("Atmosphere") then
                    fpsBackup[e] = {Density = e.Density}
                    e.Density = 0
                end
            end)
        end
    end
    
    -- Level 3: Partikel + Texturen (ASYNC - kein Thread Block!)
    if CFG.FPSLevel >= 3 then
        task.spawn(function()
            local i = 0
            for _, o in ipairs(WS:GetDescendants()) do
                if not fpsRunning then break end
                i = i + 1
                -- Alle 500 Objekte kurz yielden
                if i % 500 == 0 then task.wait() end
                
                pcall(function()
                    if o:IsA("ParticleEmitter") or o:IsA("Trail")
                    or o:IsA("Fire") or o:IsA("Smoke")
                    or o:IsA("Sparkles") or o:IsA("Beam") then
                        fpsBackup[o] = {Enabled = o.Enabled}
                        o.Enabled = false
                    elseif o:IsA("Decal") or o:IsA("Texture") then
                        fpsBackup[o] = {Transparency = o.Transparency}
                        o.Transparency = 1
                    elseif o:IsA("PointLight") or o:IsA("SpotLight") or o:IsA("SurfaceLight") then
                        fpsBackup[o] = {Enabled = o.Enabled}
                        o.Enabled = false
                    end
                end)
            end
        end)
    end
    
    notify("FPS Boost", "Level "..CFG.FPSLevel.." aktiv", 2, Color3.fromRGB(255, 200, 50))
end

local function disableFPS()
    ST.fpsboost  = false
    fpsRunning   = false
    
    -- Alles zurücksetzen
    for obj, data in pairs(fpsBackup) do
        pcall(function()
            if obj and obj.Parent then
                for k, v in pairs(data) do
                    obj[k] = v
                end
            end
        end)
    end
    fpsBackup = {}
    
    pcall(function()
        settings().Rendering.QualityLevel = Enum.QualityLevel.Automatic
        if not ST.fullbright then
            Lighting.GlobalShadows = Orig.GlobalShadows
            Lighting.FogEnd   = Orig.FogEnd
            Lighting.FogStart = Orig.FogStart
        end
    end)
    
    notify("FPS Boost", "Zurückgesetzt", 2)
end

local fpsLabel = nil
local fpsConn  = nil
local fpsC, fpsT = 0, tick()

local function enableFPSC(parent)
    ST.fpscounter = true
    if fpsLabel then pcall(function() fpsLabel:Destroy() end) end
    if fpsConn  then pcall(function() fpsConn:Disconnect() end) end
    
    fpsLabel = Instance.new("TextLabel", parent)
    fpsLabel.Size = UDim2.new(0, 90, 0, 22)
    fpsLabel.Position = UDim2.new(0, 5, 0, 5)
    fpsLabel.BackgroundColor3 = Color3.fromRGB(0,0,0)
    fpsLabel.BackgroundTransparency = 0.4
    fpsLabel.Text = "⚡ --"
    fpsLabel.TextColor3 = Color3.fromRGB(0, 255, 100)
    fpsLabel.TextSize = 12
    fpsLabel.Font = Enum.Font.Code
    fpsLabel.BorderSizePixel = 0
    fpsLabel.ZIndex = 10
    Instance.new("UICorner", fpsLabel).CornerRadius = UDim.new(0, 5)
    
    fpsC = 0
    fpsT = tick()
    
    fpsConn = RS.RenderStepped:Connect(function()
        fpsC = fpsC + 1
        local now = tick()
        if now - fpsT >= 0.5 then
            local fps = math.floor(fpsC / (now - fpsT))
            if fpsLabel and fpsLabel.Parent then
                fpsLabel.Text = "⚡ " .. fps
                fpsLabel.TextColor3 =
                    fps >= 50 and Color3.fromRGB(0, 255, 100) or
                    fps >= 30 and Color3.fromRGB(255, 255, 0)  or
                    Color3.fromRGB(255, 50, 50)
            end
            fpsC = 0
            fpsT = now
        end
    end)
end

local function disableFPSC()
    ST.fpscounter = false
    if fpsConn  then pcall(function() fpsConn:Disconnect() end); fpsConn  = nil end
    if fpsLabel then pcall(function() fpsLabel:Destroy()   end); fpsLabel = nil end
end

            if not lockedTarget
            or not lockedTarget.part
            or not pcall(function() return lockedTarget.part.Parent end) then
                lockedTarget = getBest()

            if not lockedTarget then return end
            
            -- Prüfen ob noch lebendig

            if not hum or hum.Health <= 0 then
                lockedTarget = nil
                return

            local part = lockedTarget.model:FindFirstChild(CFG.AimPart)
                      or lockedTarget.model:FindFirstChild("Head")
            if not part or not part.Parent then
                lockedTarget = nil
                return
            end
            
            local tp = part.Position
            pcall(function()
                tp = tp + part.AssemblyLinearVelocity * CFG.Prediction
            end)
            
            local goal   = CFrame.lookAt(Camera.CFrame.Position, tp)
            local smooth = math.clamp(1 - CFG.AimSmooth, 0.05, 1)
            Camera.CFrame = Camera.CFrame:Lerp(goal, smooth)

    local function switchTab(name)
        curTab = name
        for n, btn in pairs(tabBtns) do
            local on = (n == name)
            tw(btn, {
                BackgroundTransparency = on and 0 or 1,
                BackgroundColor3       = C.surf2,
            })
            local ico = btn:FindFirstChild("Ico")
            local lbl = btn:FindFirstChild("Lbl")
            if ico then tw(ico, {TextColor3 = on and C.accent or C.dim}) end
            if lbl then tw(lbl, {TextColor3 = on and C.white  or C.dim}) end
        end
        for n, pg in pairs(tabPages) do pg.Visible = (n == name) end
    end

UIS.InputBegan:Connect(function(inp, gp)
    if gp then return end
    if UIS:GetFocusedTextBox() then return end
    local k = inp.KeyCode
    
    if k == Enum.KeyCode.T then
        UI.win.Visible = not UI.win.Visible
    elseif k == Enum.KeyCode.I then UI.tg.fb()
    elseif k == Enum.KeyCode.U then UI.tg.esp()
    elseif k == Enum.KeyCode.H then UI.tg.hb()
    elseif k == Enum.KeyCode.J then UI.tg.aim()
    elseif k == Enum.KeyCode.K then UI.tg.as()
    elseif k == Enum.KeyCode.F then UI.tg.fly()
    elseif k == Enum.KeyCode.P then UI.tg.fps()
    elseif k == Enum.KeyCode.O then toggleVM()
    elseif k == Enum.KeyCode.F4 then panicAll()
    end
end)

    task.wait(1.5)
    -- Fly neu starten
    if ST.fly then
        cleanFly()
        task.wait(0.2)
        enableFly()
    end
    -- ESP refreshen
    if ST.esp then
        task.wait(0.5)
        for _, e in ipairs(getEnemies()) do applyESP(e) end
    end
    -- Hitbox refreshen
    if ST.hitbox then
        task.wait(0.5)
        for _, e in ipairs(getEnemies()) do applyHB(e) end
    end
end)
