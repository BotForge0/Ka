local Conns = {}

local function addC(tag, conn)
    Conns[tag] = Conns[tag] or {}
    table.insert(Conns[tag], conn)
end

local function clearC(tag)
    if not Conns[tag] then return end
    for _, c in ipairs(Conns[tag]) do
        pcall(function() c:Disconnect() end)
    end
    Conns[tag] = {}
end

--  DRAWING CACHE (FIX FPS)
-- ══════════════

-- Anstatt jedes Frame neue Drawing-Objekte zu erstellen,
-- erstellen wir sie einmal und verstecken sie wenn nicht gebraucht.

local DrawCache = {}

local function getDrawing(category, index, dtype)
    DrawCache[category] = DrawCache[category] or {}
    if not DrawCache[category][index] then
        if not Drawing then return nil end
        local d = Drawing.new(dtype)
        d.Visible = false
        DrawCache[category][index] = d
    end
    return DrawCache[category][index]
end

local function hideCategory(category)
    if not DrawCache[category] then return end
    for _, d in pairs(DrawCache[category]) do
        pcall(function() d.Visible = false end)
    end
end

local function clearCategory(category)
    if not DrawCache[category] then return end
    for _, d in pairs(DrawCache[category]) do
        pcall(function() d:Remove() end)
    end
    DrawCache[category] = {}
end

-- ══════════════
--  ENEMY DETECTION
-- ══════════════

local function isEnemy(obj)
    if not obj or not obj:IsA("Model") then return false end
    if obj == LP.Character then return false end
    local hum = obj:FindFirstChildOfClass("Humanoid")
    if not hum or hum.Health <= 0 then return false end
    local head = obj:FindFirstChild("Head") or obj:FindFirstChild("HumanoidRootPart")
    if not head then return false end
    return true
end

local function getEnemies()
    local list = {}
    for _, p in ipairs(Services.Players:GetPlayers()) do
        if p ~= LP and p.Character and isEnemy(p.Character) then
            table.insert(list, p.Character)
        end
    end
    -- Auch NPC-Charaktere (nicht Player)
    for _, obj in ipairs(Services.WS:GetChildren()) do
        if obj:IsA("Model") and isEnemy(obj) then
            local isPlayer = false
            for _, pl in ipairs(Services.Players:GetPlayers()) do
                if pl.Character == obj then isPlayer = true; break end
            end
            if not isPlayer then table.insert(list, obj) end
        end
    end
    return list
end

local function getBestTarget()
    local best, bestDist = nil, CFG.FOV
    local mx, my = Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2

    for _, enemy in ipairs(getEnemies()) do
        local part = enemy:FindFirstChild(CFG.AimPart)
            or enemy:FindFirstChild("Head")
            or enemy:FindFirstChild("HumanoidRootPart")
        if part then
            local tp = part.Position
            pcall(function()
                tp = tp + part.AssemblyLinearVelocity * CFG.Prediction
            end)
            local sp, vis = Camera:WorldToScreenPoint(tp)
            if vis then
                local dx = sp.X - mx
                local dy = sp.Y - my
                local dist = math.sqrt(dx*dx + dy*dy)
                if dist < bestDist then
                    bestDist = dist
                    best = {model = enemy, part = part, pos = tp, screenDist = dist}
                end
            end
        end
    end
    return best
end

local VMouseEnabled = false

local function createVirtualMouse(parent)
    if VMouseFrame then VMouseFrame:Destroy() end

    VMouseFrame = Instance.new("Frame")
    VMouseFrame.Name = "VirtualMouse"
    VMouseFrame.Size = UDim2.new(0, 20, 0, 28)
    VMouseFrame.BackgroundTransparency = 1
    VMouseFrame.ZIndex = 1000
    VMouseFrame.Visible = false
    VMouseFrame.Parent = parent

    -- Maus-Cursor Form
    local arrow = Instance.new("TextLabel")
    arrow.Size = UDim2.new(1, 0, 1, 0)
    arrow.BackgroundTransparency = 1
    arrow.Text = "➤"
    arrow.TextColor3 = Color3.fromRGB(255, 255, 255)
    arrow.TextSize = 20
    arrow.Font = Enum.Font.GothamBold
    arrow.ZIndex = 1001
    arrow.TextXAlignment = Enum.TextXAlignment.Left
    arrow.TextYAlignment = Enum.TextYAlignment.Top
    arrow.Parent = VMouseFrame

    -- Schatten
    local shadow = Instance.new("TextLabel")
    shadow.Size = UDim2.new(1, 0, 1, 0)
    shadow.Position = UDim2.new(0, 1, 0, 1)
    shadow.BackgroundTransparency = 1
    shadow.Text = "➤"
    shadow.TextColor3 = Color3.fromRGB(0, 0, 0)
    shadow.TextTransparency = 0.5
    shadow.TextSize = 20
    shadow.Font = Enum.Font.GothamBold
    shadow.ZIndex = 1000
    shadow.TextXAlignment = Enum.TextXAlignment.Left
    shadow.TextYAlignment = Enum.TextYAlignment.Top
    shadow.Parent = VMouseFrame
end

local function enableVirtualMouse()
    VMouseEnabled = true
    if VMouseFrame then VMouseFrame.Visible = true end

    -- Maus-Bewegung via Gamepad/Tastatur simulieren
    -- Pfeiltasten oder IJKL bewegen die virtuelle Maus
    local speed = 8

    if VMouseConn then VMouseConn:Disconnect() end
    VMouseConn = Services.RS.RenderStepped:Connect(function(dt)
        if not VMouseEnabled then return end

        local moved = false
        -- Bewege virtuelle Maus mit Pfeiltasten
        if Services.UIS:IsKeyDown(Enum.KeyCode.Left)  then VMousePos = VMousePos + Vector2.new(-speed, 0); moved = true end
        if Services.UIS:IsKeyDown(Enum.KeyCode.Right) then VMousePos = VMousePos + Vector2.new( speed, 0); moved = true end
        if Services.UIS:IsKeyDown(Enum.KeyCode.Up)    then VMousePos = VMousePos + Vector2.new(0, -speed); moved = true end
        if Services.UIS:IsKeyDown(Enum.KeyCode.Down)  then VMousePos = VMousePos + Vector2.new(0,  speed); moved = true end

        -- Auch echte Maus-Position tracken wenn vorhanden
        local realMousePos = Vector2.new(Mouse.X, Mouse.Y)
        if realMousePos.X ~= 0 and realMousePos.Y ~= 0 then
            VMousePos = realMousePos
        end

        -- Clamp zur Bildschirmgröße
        local vp = Camera.ViewportSize
        VMousePos = Vector2.new(
            math.clamp(VMousePos.X, 0, vp.X),
            math.clamp(VMousePos.Y, 0, vp.Y)
        )

        if VMouseFrame then
            VMouseFrame.Position = UDim2.new(0, VMousePos.X - 2, 0, VMousePos.Y - 2)
        end
    end)
end

local function disableVirtualMouse()
    VMouseEnabled = false
    if VMouseFrame then VMouseFrame.Visible = false end
    if VMouseConn then VMouseConn:Disconnect(); VMouseConn = nil end
end

local fbApplying = false

local function applyFB()
    if fbApplying then return end
    fbApplying = true
    safe(function()
        local L = Services.Lighting
        L.Brightness     = CFG.Brightness
        L.ClockTime      = CFG.ClockTime
        L.Ambient        = Color3.fromRGB(178, 178, 178)
        L.OutdoorAmbient = Color3.fromRGB(178, 178, 178)
        L.GlobalShadows  = false
        L.FogStart       = 999999
        L.FogEnd         = 999999
        for _, c in ipairs(L:GetChildren()) do
            if c:IsA("Atmosphere") then c.Density = 0 end
        end
    end)
    fbApplying = false
end

local function enableFB()
    ST.fullbright = true
    applyFB()
    local props = {"Brightness","ClockTime","Ambient","OutdoorAmbient","GlobalShadows","FogStart","FogEnd"}
    for _, prop in ipairs(props) do
        addC("fb", Services.Lighting:GetPropertyChangedSignal(prop):Connect(function()
            if ST.fullbright and not fbApplying then applyFB() end
        end))
    end
    notify("Fullbright", "Aktiviert ✓", 2, Color3.fromRGB(255, 200, 50))
end

local function disableFB()
    ST.fullbright = false
    clearC("fb")
    safe(function()
        fbApplying = true
        local L = Services.Lighting
        L.Brightness     = Orig.Brightness
        L.ClockTime      = Orig.ClockTime
        L.Ambient        = Orig.Ambient
        L.OutdoorAmbient = Orig.OutdoorAmbient
        L.GlobalShadows  = Orig.GlobalShadows
        L.FogStart       = Orig.FogStart
        L.FogEnd         = Orig.FogEnd
        fbApplying = false
    end)
    notify("Fullbright", "Deaktiviert", 2)
end

local function applyESP(model)
    if not isEnemy(model) or model:FindFirstChild("BRM6_ESP") then return end
    safe(function()
        local hl = Instance.new("Highlight")
        hl.Name              = "BRM6_ESP"
        hl.Adornee           = model
        hl.FillColor         = Color3.fromRGB(255, 50, 50)
        hl.FillTransparency  = CFG.ESP_Fill
        hl.OutlineColor      = Color3.fromRGB(255, 230, 0)
        hl.OutlineTransparency = CFG.ESP_Outline
        hl.DepthMode         = Enum.HighlightDepthMode.AlwaysOnTop
        hl.Parent            = model
    end)
end

local function enableESP()
    ST.esp = true
    for _, e in ipairs(getEnemies()) do applyESP(e) end

    addC("esp", Services.WS.DescendantAdded:Connect(function(d)
        if not ST.esp then return end
        task.delay(0.5, function()
            if d and d:IsA("Model") then applyESP(d) end
        end)
    end))

    -- Refresh-Loop (langsam, kein FPS-Impact)
    task.spawn(function()
        while ST.esp do
            task.wait(4)
            if ST.esp then
                for _, e in ipairs(getEnemies()) do applyESP(e) end
            end
        end
    end)

    notify("ESP", "Highlight aktiv", 2, Color3.fromRGB(255, 255, 0))
end

local function disableESP()
    ST.esp = false
    clearC("esp")
    for _, obj in ipairs(Services.WS:GetDescendants()) do
        if obj:IsA("Highlight") and obj.Name == "BRM6_ESP" then
            pcall(function() obj:Destroy() end)
        end
    end
    notify("ESP", "Deaktiviert", 2)
end

local flyBody, flyGyro

local function enableFly()
    local ch = LP.Character
    if not ch then return end
    local root = ch:FindFirstChild("HumanoidRootPart")
    if not root then return end

    ST.fly = true

    safe(function()
        local hum = ch:FindFirstChildOfClass("Humanoid")
        if hum then hum.PlatformStand = true end

        flyBody = Instance.new("BodyVelocity")
        flyBody.MaxForce = Vector3.new(1e9, 1e9, 1e9)
        flyBody.Velocity  = Vector3.zero
        flyBody.Parent    = root

        flyGyro = Instance.new("BodyGyro")
        flyGyro.MaxTorque = Vector3.new(1e9, 1e9, 1e9)
        flyGyro.D = 200
        flyGyro.P = 40000
        flyGyro.Parent = root
    end)

    -- Noclip
    addC("fly_nc", Services.RS.Stepped:Connect(function()
        if not ST.fly then return end
        local c = LP.Character
        if not c then return end
        for _, p in ipairs(c:GetDescendants()) do
            if p:IsA("BasePart") then p.CanCollide = false end
        end
    end))

    -- Fly Movement
    addC("fly", Services.RS.RenderStepped:Connect(function()
        if not ST.fly then return end
        local c = LP.Character
        if not c then return end
        local r = c:FindFirstChild("HumanoidRootPart")
        if not r then return end

        local dir = Vector3.zero
        local cf  = Camera.CFrame
        local spd = CFG.FlySpeed
        if Services.UIS:IsKeyDown(Enum.KeyCode.LeftShift) then spd = spd * 2 end

        if Services.UIS:IsKeyDown(Enum.KeyCode.W)          then dir = dir + cf.LookVector  end
        if Services.UIS:IsKeyDown(Enum.KeyCode.S)          then dir = dir - cf.LookVector  end
        if Services.UIS:IsKeyDown(Enum.KeyCode.A)          then dir = dir - cf.RightVector end
        if Services.UIS:IsKeyDown(Enum.KeyCode.D)          then dir = dir + cf.RightVector end
        if Services.UIS:IsKeyDown(Enum.KeyCode.Space)      then dir = dir + Vector3.yAxis  end
        if Services.UIS:IsKeyDown(Enum.KeyCode.LeftControl) then dir = dir - Vector3.yAxis end

        if dir.Magnitude > 0 then dir = dir.Unit end
        if flyBody then flyBody.Velocity = dir * spd end
        if flyGyro then flyGyro.CFrame   = cf         end
    end))

    notify("Fly", "Speed: " .. CFG.FlySpeed .. " | Shift=2x", 2, Color3.fromRGB(100, 150, 255))
end

local function disableFly()
    ST.fly = false
    clearC("fly")
    clearC("fly_nc")
    safe(function()
        local ch = LP.Character
        if ch then
            local h = ch:FindFirstChildOfClass("Humanoid")
            if h then h.PlatformStand = false end
        end
        if flyBody then flyBody:Destroy() end
        if flyGyro  then flyGyro:Destroy()  end
    end)
    flyBody = nil
    flyGyro = nil
    notify("Fly", "Deaktiviert", 2)
end

local fpsBackup = {}

local function enableFPS()
    ST.fpsboost = true
    safe(function()
        settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
    end)
    safe(function()
        Services.Lighting.GlobalShadows = false
        Services.Lighting.FogEnd   = 9e9
        Services.Lighting.FogStart = 9e9
        local t = Services.WS:FindFirstChildOfClass("Terrain")
        if t then
            t.WaterWaveSize   = 0
            t.WaterWaveSpeed  = 0
            t.WaterReflectance = 0
        end
    end)

    if CFG.FPSLevel >= 2 then
        for _, e in ipairs(Services.Lighting:GetDescendants()) do
            safe(function()
                if e:IsA("PostEffect") then
                    fpsBackup[e] = {Enabled = e.Enabled}
                    e.Enabled = false
                elseif e:IsA("Atmosphere") then
                    fpsBackup[e] = {Density = e.Density}
                    e.Density = 0
                end
            end)
        end
    end

    if CFG.FPSLevel >= 3 then
        local count = 0
        for _, o in ipairs(Services.WS:GetDescendants()) do
            count = count + 1
            if count % 3000 == 0 then task.wait() end
            safe(function()
                if o:IsA("ParticleEmitter") or o:IsA("Trail")
                or o:IsA("Beam") or o:IsA("Fire")
                or o:IsA("Smoke") or o:IsA("Sparkles") then
                    fpsBackup[o] = {Enabled = o.Enabled}
                    o.Enabled = false
                elseif o:IsA("Decal") or o:IsA("Texture") then
                    fpsBackup[o] = {Transparency = o.Transparency}
                    o.Transparency = 1
                end
            end)
        end
    end

    notify("FPS Boost", "Level " .. CFG.FPSLevel, 2, Color3.fromRGB(255, 200, 50))
end

local function disableFPS()
    ST.fpsboost = false
    for obj, data in pairs(fpsBackup) do
        safe(function()
            if obj and obj.Parent then
                for k, v in pairs(data) do obj[k] = v end
            end
        end)
    end
    fpsBackup = {}
    safe(function()
        settings().Rendering.QualityLevel = Enum.QualityLevel.Automatic
        if not ST.fullbright then
            Services.Lighting.GlobalShadows = Orig.GlobalShadows
        end
    end)
    notify("FPS Boost", "Deaktiviert", 2)
end

                           or lockedTarget.model:FindFirstChild("Head")

                    pcall(function() tp = tp + part.AssemblyLinearVelocity * CFG.Prediction end)
                    local goal = CFrame.lookAt(Camera.CFrame.Position, tp)
                    local smooth = math.clamp(1 - CFG.AimSmooth, 0.05, 1)
                    Camera.CFrame = Camera.CFrame:Lerp(goal, smooth)

                        else

local fpsLabel  = nil
local fpsCount  = 0
local fpsLast   = tick()
local fpsMaster = nil

local function enableFPSC(parent)
    ST.fpscounter = true
    if fpsLabel then pcall(function() fpsLabel:Destroy() end) end

    fpsLabel = Instance.new("TextLabel")
    fpsLabel.Size                  = UDim2.new(0, 100, 0, 24)
    fpsLabel.Position              = UDim2.new(0, 6, 0, 6)
    fpsLabel.BackgroundColor3      = Color3.fromRGB(0,0,0)
    fpsLabel.BackgroundTransparency = 0.45
    fpsLabel.Text                  = "⚡ --"
    fpsLabel.TextColor3            = Color3.fromRGB(0, 255, 100)
    fpsLabel.TextSize              = 13
    fpsLabel.Font                  = Enum.Font.Code
    fpsLabel.BorderSizePixel       = 0
    fpsLabel.ZIndex                = 10
    fpsLabel.Parent                = parent
    Instance.new("UICorner", fpsLabel).CornerRadius = UDim.new(0, 6)

    fpsCount = 0
    fpsLast  = tick()

    fpsMaster = Services.RS.RenderStepped:Connect(function()
        fpsCount = fpsCount + 1
        local now = tick()
        if now - fpsLast >= 0.5 then
            local fps = math.floor(fpsCount / (now - fpsLast))
            if fpsLabel then
                fpsLabel.Text = "⚡ " .. fps .. " FPS"
                fpsLabel.TextColor3 = fps >= 50 and Color3.fromRGB(0,255,100)
                    or fps >= 30 and Color3.fromRGB(255,255,0)
                    or Color3.fromRGB(255,50,50)
            end
            fpsCount = 0
            fpsLast  = now
        end
    end)
end

local function disableFPSC()
    ST.fpscounter = false
    if fpsMaster then pcall(function() fpsMaster:Disconnect() end); fpsMaster = nil end
    if fpsLabel   then pcall(function() fpsLabel:Destroy()  end); fpsLabel  = nil end
end

        end)

            if dragging and inp.UserInputType == Enum.UserInputType.MouseMovement then
                updateSlider(inp.Position.X)
            end
        end)
        Services.UIS.InputEnded:Connect(function(inp)
            if inp.UserInputType == Enum.UserInputType.MouseButton1 then
                dragging = false
            end
        end)
    end

-- ══════════════
--  KEYBINDS
-- ══════════════

Services.UIS.InputBegan:Connect(function(inp, gp)
    if gp then return end
    if Services.UIS:GetFocusedTextBox() then return end

    local key = inp.KeyCode

    if key == Enum.KeyCode.T then
        UI.main.Visible = not UI.main.Visible

    elseif key == Enum.KeyCode.I then
        UI.toggleFuncs.fullbright()

    elseif key == Enum.KeyCode.U then
        UI.toggleFuncs.esp()

    elseif key == Enum.KeyCode.H then
        UI.toggleFuncs.hitbox()

    elseif key == Enum.KeyCode.J then
        UI.toggleFuncs.aimbot()

    elseif key == Enum.KeyCode.K then
        UI.toggleFuncs.autoshoot()

    elseif key == Enum.KeyCode.F then
        UI.toggleFuncs.fly()

    elseif key == Enum.KeyCode.P then
        UI.toggleFuncs.fpsboost()

    elseif key == Enum.KeyCode.O then
        -- Virtuelle Maus togglen
        VMouseEnabled = not VMouseEnabled
        if VMouseEnabled then
            enableVirtualMouse()
            notify("Virtuelle Maus", "AN - Pfeiltasten bewegen", 2, C.cyan or Color3.fromRGB(40,195,255))
        else
            disableVirtualMouse()
            notify("Virtuelle Maus", "AUS", 2)
        end

    elseif key == Enum.KeyCode.F4 then
        panicAll()

    -- Virtuelle Maus Klick
    elseif key == Enum.KeyCode.Return and VMouseEnabled then
        -- Simuliere Linksklick an virtueller Maus Position
        -- (Funktioniert über das Hover-System)
        notify("VMaus Klick", "Enter = Klick simuliert", 1)
    end
end)

-- ══════════════
--  RESPAWN
-- ══════════════

LP.CharacterAdded:Connect(function()
    task.wait(1.5)
    if ST.fly then
        disableFly()
        task.wait(0.3)
        enableFly()
    end
    if ST.esp then
        task.spawn(function()
            task.wait(0.5)
            for _, e in ipairs(getEnemies()) do applyESP(e) end
        end)
    end
    if ST.hitbox then
        task.spawn(function()
            task.wait(0.5)
            for _, e in ipairs(getEnemies()) do applyHB(e) end
        end)
    end
end)

LP.CharacterRemoving:Connect(function()
    if flyBody then pcall(function() flyBody:Destroy() end); flyBody = nil end
    if flyGyro  then pcall(function() flyGyro:Destroy()  end); flyGyro  = nil end
end)
