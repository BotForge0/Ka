function isEnemy(obj)
    if not obj or not obj:IsA("Model") then return false end
    if obj == LP.Character then return false end

    local hum = obj:FindFirstChildOfClass("Humanoid")
    if not hum or hum.Health <= 0 then return false end

    local head = obj:FindFirstChild("Head")
    local root = obj:FindFirstChild("HumanoidRootPart")
    if not head and not root then return false end

    -- Bodypart check
    local parts = {"Left Arm","Right Arm","Left Leg","Right Leg","Torso",
                    "LeftHand","RightHand","LeftFoot","RightFoot","UpperTorso",
                    "LeftUpperArm","RightUpperArm","LeftUpperLeg","RightUpperLeg"}
    for _, pn in ipairs(parts) do
        if obj:FindFirstChild(pn) then return true end
    end
    return false
end

local function getEnemies()
    local e = {}
    for _, o in pairs(WS:GetDescendants()) do
        if isEnemy(o) then table.insert(e, o) end
    end
    return e
end

local function getTargetFOV()
    local best, bestD = nil, CFG.FOV
    for _, e in pairs(getEnemies()) do
        local part = e:FindFirstChild(CFG.AimPart) or e:FindFirstChild("Head") or e:FindFirstChild("HumanoidRootPart")
        if part then
            local tp = part.Position
            pcall(function() tp = tp + part.AssemblyLinearVelocity * CFG.Prediction end)
            local sp, vis = Camera:WorldToScreenPoint(tp)
            if vis then
                local d = (Vector2.new(Mouse.X, Mouse.Y + 36) - Vector2.new(sp.X, sp.Y)).Magnitude
                if d < bestD then
                    bestD = d
                    best = {model = e, part = part, pos = tp, dist = d}
                end
            end
        end
    end
    return best
end

local function enableLootESP()
    ST.lootESP = true
    if not Drawing then ST.lootESP = false return end

    local lootNames = {"gun","rifle","pistol","ammo","medkit","armor","grenade","magazine",
                        "attachment","scope","suppressor","grip","barrel","weapon","loot",
                        "item","pickup","drop","crate","supply","bandage","health","vest",
                        "helmet","backpack","ak","m4","m16","scar","sniper","shotgun","smg",
                        "rpg","knife","melee"}

    local function isLoot(obj)
        if not obj:IsA("Model") and not obj:IsA("BasePart") and not obj:IsA("Tool") then return false end
        if obj:FindFirstChildOfClass("Humanoid") then return false end
        local n = obj.Name:lower()
        for _, ln in ipairs(lootNames) do
            if n:find(ln) then return true end
        end
        -- Check for ClickDetector or ProximityPrompt
        if obj:FindFirstChildOfClass("ClickDetector") or obj:FindFirstChildOfClass("ProximityPrompt") then
            return true
        end
        return false
    end

    addC("lootESP", RS.RenderStepped:Connect(function()
        if not ST.lootESP then return end
        clearLootESP()

        local myRoot = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")

        for _, obj in pairs(WS:GetDescendants()) do
            if isLoot(obj) then
                local pos = nil
                if obj:IsA("Model") then
                    local prim = obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")
                    if prim then pos = prim.Position end
                elseif obj:IsA("BasePart") then
                    pos = obj.Position
                end

                if pos then
                    local sp, vis = Camera:WorldToScreenPoint(pos)
                    if vis then
                        local dist = myRoot and math.floor((myRoot.Position - pos).Magnitude) or 0
                        if dist < 300 then
                            local drawings = {}

                            local txt = Drawing.new("Text")
                            txt.Text = obj.Name .. " [" .. dist .. "m]"
                            txt.Position = Vector2.new(sp.X, sp.Y)
                            txt.Color = Color3.fromRGB(50, 200, 255)
                            txt.Size = 12
                            txt.Center = true
                            txt.Outline = true
                            txt.OutlineColor = Color3.fromRGB(0, 0, 0)
                            txt.Font = 2
                            txt.Visible = true
                            table.insert(drawings, txt)

                            lootDrawings[obj] = drawings
                        end
                    end
                end
            end
        end
    end))

    notify("Loot ESP", "Items werden angezeigt", 2, Color3.fromRGB(50, 200, 255))
end

    ST.lootESP = false
    clearC("lootESP")

            fovCircle.Color = CFG.FOV_Color
            fovCircle.Transparency = 0.7
            fovCircle.Visible = CFG.FOV_Visible
        end
    end)

    addC("aim", RS.RenderStepped:Connect(function()
        if not ST.aimbot then return end
        if fovCircle then
            safe(function()
                fovCircle.Position = Vector2.new(Mouse.X, Mouse.Y + 36)
                fovCircle.Radius = CFG.FOV
                fovCircle.Color = CFG.FOV_Color
                fovCircle.Visible = CFG.FOV_Visible and ST.aimbot
            end)
        end
    end))

    addC("aim", RS.Heartbeat:Connect(function()
        if not ST.aimbot then return end
        if not UIS:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
            lockedTarget = nil
            return
        end

        if not lockedTarget or not lockedTarget.part or not lockedTarget.part.Parent then
            lockedTarget = getTargetFOV()
            if not lockedTarget then return end
        end

        local hum = lockedTarget.model:FindFirstChildOfClass("Humanoid")
        if not hum or hum.Health <= 0 then
            lockedTarget = nil
            return
        end

        local aimPart = lockedTarget.model:FindFirstChild(CFG.AimPart) or lockedTarget.model:FindFirstChild("Head")
        if not aimPart then lockedTarget = nil return end

        local tp = aimPart.Position
        pcall(function() tp = tp + aimPart.AssemblyLinearVelocity * CFG.Prediction end)

        local targetCF = CFrame.lookAt(Camera.CFrame.Position, tp)
        local smooth = math.clamp(1 - CFG.AimSmooth, 0.05, 1)
        Camera.CFrame = Camera.CFrame:Lerp(targetCF, smooth)
    end))

    notify("Aimbot", "FOV: "..CFG.FOV.."px | Smooth: "..math.floor(CFG.AimSmooth*100).."%", 2, Color3.fromRGB(255, 100, 100))

local function clearCrosshair()
    for _, l in pairs(crosshairLines) do
        pcall(function() l:Remove() end)
    end
    crosshairLines = {}

local function enableCrosshair()
    ST.crosshair = true
    if not Drawing then ST.crosshair = false return end

    addC("crosshair", RS.RenderStepped:Connect(function()
        if not ST.crosshair then return end
        clearCrosshair()

        local cx = Camera.ViewportSize.X / 2
        local cy = Camera.ViewportSize.Y / 2
        local s = CFG.CrosshairSize
        local g = CFG.CrosshairGap

        local dirs = {
            {Vector2.new(cx - g, cy), Vector2.new(cx - g - s, cy)},
            {Vector2.new(cx + g, cy), Vector2.new(cx + g + s, cy)},
            {Vector2.new(cx, cy - g), Vector2.new(cx, cy - g - s)},
            {Vector2.new(cx, cy + g), Vector2.new(cx, cy + g + s)},
        }

        for _, d in ipairs(dirs) do
            local line = Drawing.new("Line")
            line.From = d[1]
            line.To = d[2]
            line.Color = CFG.CrosshairColor
            line.Thickness = CFG.CrosshairThickness
            line.Visible = true
            table.insert(crosshairLines, line)
        end

        -- Center dot
        local dot = Drawing.new("Circle")
        dot.Position = Vector2.new(cx, cy)
        dot.Radius = 2
        dot.Color = CFG.CrosshairColor
        dot.Filled = true
        dot.Visible = true
        table.insert(crosshairLines, dot)
    end))

    notify("Crosshair", "Custom Crosshair aktiviert", 2, Color3.fromRGB(0, 255, 0))
end

        flyGyro.P = 40000
        flyGyro.Parent = r
    end)

        if not ST.fly or not r or not r.Parent then return end
        local d = Vector3.zero
        local cf = Camera.CFrame
        local sp = CFG.FlySpeed
        if UIS:IsKeyDown(Enum.KeyCode.LeftShift) then sp = sp * 2 end
        if UIS:IsKeyDown(Enum.KeyCode.W) then d = d + cf.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.S) then d = d - cf.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.A) then d = d - cf.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.D) then d = d + cf.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.Space) then d = d + Vector3.yAxis end
        if UIS:IsKeyDown(Enum.KeyCode.LeftControl) then d = d - Vector3.yAxis end
        if d.Magnitude > 0 then d = d.Unit end
        if flyBody then flyBody.Velocity = d * sp end
        if flyGyro then flyGyro.CFrame = cf end
    end))

    notify("Fly", "Speed: "..CFG.FlySpeed.." | Shift = 2x", 2, Color3.fromRGB(100, 150, 255))
end

-- ═══════════════════
--  FPS BOOST (GEFIXT)
-- ═══════════════════

local function enableFPS()
    ST.fpsboost = true
    safe(function() settings().Rendering.QualityLevel = Enum.QualityLevel.Level01 end)
    safe(function()
        Lighting.GlobalShadows = false
        Lighting.FogEnd = 9e9
        Lighting.FogStart = 9e9
        pcall(function() MS.Use2022Materials = false end)
        local t = WS:FindFirstChildOfClass("Terrain")
        if t then
            t.WaterWaveSize = 0
            t.WaterWaveSpeed = 0
            t.WaterReflectance = 0
            t.WaterTransparency = 0
            pcall(function() sethiddenproperty(t, "Decoration", false) end)
        end
    end)
    if CFG.FPSLevel >= 2 then
        for _, e in pairs(Lighting:GetDescendants()) do
            safe(function()
                if e:IsA("PostEffect") then potatoOrig[e] = {Enabled = e.Enabled}; e.Enabled = false end
                if e:IsA("Atmosphere") then potatoOrig[e] = {Density = e.Density}; e.Density = 0 end
            end)
        end
    end
    if CFG.FPSLevel >= 3 then
        local c = 0
        for _, o in pairs(WS:GetDescendants()) do
            c = c + 1
            if c % 4000 == 0 then task.wait() end
            safe(function()
                if o:IsA("Decal") or o:IsA("Texture") then
                    potatoOrig[o] = {Transparency = o.Transparency}; o.Transparency = 1
                elseif o:IsA("ParticleEmitter") or o:IsA("Trail") or o:IsA("Beam") or o:IsA("Fire") or o:IsA("Smoke") or o:IsA("Sparkles") then
                    potatoOrig[o] = {Enabled = o.Enabled}; o.Enabled = false
                elseif o:IsA("MeshPart") then
                    potatoOrig[o] = {Material = o.Material, TextureID = o.TextureID}
                    o.Material = Enum.Material.SmoothPlastic; o.TextureID = ""
                elseif o:IsA("BasePart") and not o:IsA("MeshPart") and not o:IsA("Terrain") then
                    potatoOrig[o] = {Material = o.Material}; o.Material = Enum.Material.SmoothPlastic
                elseif o:IsA("PointLight") or o:IsA("SpotLight") or o:IsA("SurfaceLight") then
                    potatoOrig[o] = {Shadows = o.Shadows}; o.Shadows = false
                end
            end)
        end
    end
    notify("FPS Boost", "Level "..CFG.FPSLevel.." aktiviert", 2, Color3.fromRGB(255, 200, 50))
end

local function disableFPS()
    ST.fpsboost = false
    for o, d in pairs(potatoOrig) do
        safe(function()
            if o and o.Parent then
                for p, v in pairs(d) do o[p] = v end
            end
        end)
    end
    potatoOrig = {}
    safe(function() settings().Rendering.QualityLevel = Enum.QualityLevel.Automatic end)
    if not ST.fullbright then
        safe(function() Lighting.GlobalShadows = Orig.GlobalShadows end)
    end
    notify("FPS Boost", "Deaktiviert", 2)
end

local function enableFPSC(parent)
    ST.fpscounter = true
    if fpsLabel then pcall(function() fpsLabel:Destroy() end) end
    fpsLabel = Instance.new("TextLabel")
    fpsLabel.Size = UDim2.new(0, 110, 0, 26)
    fpsLabel.Position = UDim2.new(0, 8, 0, 5)
    fpsLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    fpsLabel.BackgroundTransparency = 0.4
    fpsLabel.Text = "FPS: --"
    fpsLabel.TextColor3 = Color3.fromRGB(0, 255, 100)
    fpsLabel.TextSize = 14
    fpsLabel.Font = Enum.Font.Code
    fpsLabel.BorderSizePixel = 0
    fpsLabel.Parent = parent
    Instance.new("UICorner", fpsLabel).CornerRadius = UDim.new(0, 6)
    local fc, lt = 0, tick()
    fpsConn = RS.RenderStepped:Connect(function()
        fc = fc + 1
        local n = tick()
        if n - lt >= 0.5 then
            local fps = math.floor(fc / (n - lt))
            if fpsLabel then
                fpsLabel.Text = " ⚡ " .. fps .. " FPS"
                fpsLabel.TextColor3 = fps >= 50 and Color3.fromRGB(0, 255, 100)
                    or fps >= 30 and Color3.fromRGB(255, 255, 0)
                    or Color3.fromRGB(255, 50, 50)
            end
            fc = 0
            lt = n
        end
    end)
end

local function disableFPSC()
    ST.fpscounter = false
    if fpsConn then pcall(function() fpsConn:Disconnect() end) end
    if fpsLabel then pcall(function() fpsLabel:Destroy() end) fpsLabel = nil end
end

                        CFG[k] = v
                    end

            end
            notify("Config", "Geladen!", 2, Color3.fromRGB(0, 255, 100))
        end
    end)

    local tBtns, tPages, curTab = {}, {}, nil

    local function switchTab(name)
        curTab = name
        for n, b in pairs(tBtns) do
            local icon = b:FindFirstChild("Ico")
            if n == name then
                tw(b, {BackgroundColor3 = P.surface2, BackgroundTransparency = 0})
                if icon then tw(icon, {TextColor3 = P.accent}) end
                for _, c in pairs(b:GetChildren()) do
                    if c:IsA("TextLabel") and c.Name ~= "Ico" then tw(c, {TextColor3 = P.white}) end
                end
            else
                tw(b, {BackgroundTransparency = 1})
                if icon then tw(icon, {TextColor3 = P.dim}) end
                for _, c in pairs(b:GetChildren()) do
                    if c:IsA("TextLabel") and c.Name ~= "Ico" then tw(c, {TextColor3 = P.dim}) end
                end
            end
        end
        for n, p in pairs(tPages) do p.Visible = (n == name) end
    end

UIS.InputBegan:Connect(function(i, gp)
    if gp or UIS:GetFocusedTextBox() then return end
    if i.KeyCode == Enum.KeyCode.I then UI.toggles.fullbright()
    elseif i.KeyCode == Enum.KeyCode.U then UI.toggles.esp()
    elseif i.KeyCode == Enum.KeyCode.H then UI.toggles.hitbox()
    elseif i.KeyCode == Enum.KeyCode.J then UI.toggles.aimbot()
    elseif i.KeyCode == Enum.KeyCode.K then UI.toggles.autoshoot()
    elseif i.KeyCode == Enum.KeyCode.F then UI.toggles.fly()
    elseif i.KeyCode == Enum.KeyCode.P then UI.toggles.fpsboost()
    elseif i.KeyCode == Enum.KeyCode.T then UI.main.Visible = not UI.main.Visible
    elseif i.KeyCode == Enum.KeyCode.F4 then panicDisableAll()
    end
end)
