local keyNames = {}
for _, kc in pairs(Enum.KeyCode:GetEnumItems()) do
 local s = tostring(kc):gsub("Enum.KeyCode.", "")
 if #s <= 6 then keyNames[kc] = s end
end
keyNames[Enum.KeyCode.Zero] = "0"
keyNames[Enum.KeyCode.LeftShift] = "Shift"
keyNames[Enum.KeyCode.LeftControl] = "Ctrl"
keyNames[Enum.KeyCode.LeftAlt] = "Alt"
keyNames[Enum.KeyCode.Space] = "Space"

local function keyName(kc)
 return keyNames[kc] or tostring(kc):gsub("Enum.KeyCode.",""):sub(1,6)
end

local function makeKeyBtn(parent, actionId)
 local saved = CFG.Keybinds[actionId]
 local btnText = "⚡ Key: " .. (saved and keyName(saved) or "?")
 if IS_MOBILE then
 btnText = "⚡ Touch: " .. (saved and "✅" or "❌")
 end
 local btn = parent:AddButton(btnText, function()
 if IS_MOBILE then
 \-- Mobile: toggle on/off
 local current = CFG.Keybinds[actionId]
 if current then
 CFG.Keybinds[actionId] = nil
 btn.Text = "⚡ Touch: ❌"
 else
 CFG.Keybinds[actionId] = true
 btn.Text = "⚡ Touch: ✅"
 end
 save()
 else
 \-- PC: keybind listener
 local listening = true
 btn.Text = "⚡ Drücke Taste..."
 local conn = C_UIS.InputBegan:Connect(function(inp, gp)
 if gp then return end
 if inp.UserInputType == Enum.UserInputType.Keyboard then
 local kc = inp.KeyCode
 if kc ~= Enum.KeyCode.Unknown and kc ~= Enum.KeyCode.Zero then
 CFG.Keybinds[actionId] = kc; save()
 btn.Text = "⚡ Key: " .. keyName(kc)
 if conn then conn:Disconnect() end
 end
 end
 end)
 task.delay(4, function()
 if listening then
 local s = CFG.Keybinds[actionId]
 btn.Text = "⚡ Key: " .. (s and keyName(s) or "?")
 if conn then conn:Disconnect() end
 end
 end)
 end
 end)
 return btn
end
